#include  <stdio.h>
int hour, minute, second; 

struct clock
{  int hour;
int minute;
int second; };
typedef struct clock CLOCK;








void Update(CLOCK *a)                  
{	a->second++;
	if (a->second == 60) 	
	{	a->second = 0;
		a->minute++;
	}  
	if (a->minute == 60)		
{a->minute = 0;
		a->hour++;
	}
	if (a->hour == 24)	
{	a->hour = 0;
	}
}

void Display(CLOCK *a)	
{	printf("%2d:%2d:%2d\r",a->hour,a->minute,a->second); 
}

//函数功能：模拟延迟1秒的时间
